```python
# 이미지 연속 캡쳐 (빠르게 사진 찍을때)

from time import sleep
from picamera import PiCamera

camera = PiCamera(resolution=(1280, 720), framerate=30) # 인스턴스 생성

camera.iso = 100
sleep(2)

camera.shutter_speed = camera.exposure_speed
camera.exposure_mode = 'off'
g = camera.awb_gains
camera.awb_mode = 'off'
camera.awb_gains = g

# 고정된 설정으로 여러 사진 찍기  파일 명에 리스트를 넘겨줌
camera.capture_sequence(['image%02d.jpg' % i for i in range(10)])
```

```python
# 천천히 사진을 찍고싶을때 continuous

from time import sleep
from picamera import PiCamera

camera = PiCamera()
camera.start_preview()
sleep(2)

for filename in camera.capture_continuous('img{counter:03d}.jpg'):
    print('Captured %s' % filename)
    sleep(10) # wait 10 seconds 간격마다 사진을 찍음
```



## 2) PiCamera를 이용한 예제

- vutil 모듈화 시키기(해상도 조정용으로 자주쓰일예정)

```python
def get_resolution():
    #해상도 선택하도록함
    res = int(input('Resolution(1:320x240, 2:640x480, 3:1024x768)?'))

    # 선택한 값에 따라 해상도 설정 (다른데서도 써먹을 수있음)
    if res == 3:
        resolution = (1024, 768)
    elif res == 2:
        resolution = (640, 480)
    else:
        resolution = (320, 240)

    return resolution
```

- 원하는 해상도로 사진찍고 저장하는 예제

```python
# picamera 라이브러리 임포트
import picamera

# time 라이브러리 임포트
import time

# PiCamera 객체 인스턴스 생성
with picamera.PiCamera() as camera:
    # 해상도를 선택하도록 함
    res = int(input('Resolution(1:320x240, 2:640x480, 3:1024x768)?'))
    
    # 선택한 값에 따라 해상도 설정 (다른데서도 써먹을 수있음)
    if res == 3:
        camera.resolution = (1024, 768)
    elif res == 2:
        camera.resolution = (640, 480)
    else:
        camera.resolution = (320, 240)
    
    # 파일명 입력 받기
    filename = input('File Name?')

    # 프리뷰 화면 표시
    camera.start_preview()

    # 1초 대기
    time.sleep(1)

    # 프리뷰 종료
    camera.stop_preview()

    # 촬영하고 저장
    camera.capture(filename + '.jpg')
```

- 비디오 녹화예제

```python
# 비디오 녹화
import picamera
import time

with picamera.PiCamera() as camera:
    res = int(raw_input('Resolution(1:320x240, 2:640x480, 3:1024x768)?'))
    # 선택한 값에 따라 해상도 설정
    if res == 3:
    	camera.resolution = (1024, 768)
    elif res == 2:
    	camera.resolution = (640, 480)
    else:
    	camera.resolution = (320, 240)

    # 파일명 입력
    filename = raw_input('File Name?')
    camera.framerate = 15

    # 프리뷰 화면 표시
    camera.start_preview()

    # 촬영하고 파일 저장
    camera.start_recording(output = filename + '.h264')
    # 5초 대기
    camera.wait_recording(5)
    # 촬영 종료
    camera.stop_recording()
    # 프리뷰 화면 종료
    camera.stop_preview()
```
